Instalação normal:
http://www.datsi.fi.upm.es/~frosal/sources/shc-3.8.9.tgz
cd ~/Downloads
tar xvfz shc-3.8.9.tgz
cd shc-3.8.9
make
sudo make install
sudo cp shc /bin/shc
sudo chmod +x /bin/shc
